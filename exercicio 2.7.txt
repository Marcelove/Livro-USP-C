#include <stdio.h>

main(){

    int a, b, c;

    printf("Digite os catetos e eu direi se é um triângulo ou não. Se for, direi qual o tipo:\n");
    scanf("%d %d %d", &a, &b, &c);

    if (a < b+c && b < a+c && c < a+b){
        if (a == b  && b ==c ){
            printf("\nÉ um triângulo equilátero.\n");
        }
        else if (a == b || a == c || b == c){
            printf("\nÉ um triângulo isósceles.\n");
        }
        else if (a != b != c){
            printf("\nÉ um triângulo escaleno.\n");
        }

    } else {
        printf("\nNão é um triângulo.\n");
    }
}
