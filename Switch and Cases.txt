#include <stdio.h>

main(){

    int n;

    printf("Digite um número:");
    scanf("%d", &n);

    switch(n){

        case 1: putchar('A'); break;
        case 2: putchar('B');
        case 3: putchar('C'); break;
        case 4: putchar('D');
        default: printf('*');
        case 5: putchar('F');

    }
    putchar('.');
}
