#include <stdio.h>

main(){

    int x, y;
    char op;

    printf("Digite a operação no formato A*B:");
    scanf("%d %c %d", &x, &op, &y);

    switch(op){
        case '*': printf("Seu resultado é %d.", x*y);
        case '/': if (y == 0){

            printf("Impossível dividir por zero!.");
            exit(1);

        }

        printf("Seu resultado é %d.", x/y);


    }



}
